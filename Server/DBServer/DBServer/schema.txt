1. CREATE DATABASE  IF NOT EXISTS `tiiiino` 

2. USE `tiiiino`;

3. DROP TABLE IF EXISTS `userinfo`;

4. CREATE TABLE `userinfo` (
  `UID` int NOT NULL AUTO_INCREMENT COMMENT 'Unique ID',
  `ID` varchar(20) NOT NULL COMMENT 'account ID',
  `PassWord` varchar(20) NOT NULL COMMENT 'account PW',
  `nick` varchar(20) NOT NULL COMMENT 'user nickname',
  `credit` decimal(2,1) NOT NULL DEFAULT '0.0' COMMENT 'mmr',
  `point` int NOT NULL DEFAULT '0',
  `state` int NOT NULL DEFAULT '0',
  PRIMARY KEY (`UID`),
  UNIQUE KEY `ID_UNIQUE` (`ID`),
  UNIQUE KEY `nick_UNIQUE` (`nick`)
) ENGINE=InnoDB AUTO_INCREMENT=9 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
